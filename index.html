Lock-in is bullshit

It's crazy hard to implement, right?


      <script src="https://login.persona.org/include.js"></script>
      <script>
      navigator.id.watch({
        onlogin: function(assertion) {
          $.post('/login', { assertion: assertion }, function() {
            document.location = document.location;
          });
        }
      });
      $(".login_button").on('click', function() { navigator.id.request(); });
      </script>

How about logout?


      <script src="https://login.persona.org/include.js"></script>
      <script>
      navigator.id.watch({
        onlogout: function() { document.location = '/logout'; }
      });

      $(".logout_button").on('click', function() { navigator.id.logout(); });
      </script>
